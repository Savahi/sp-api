<h1>Общие команды</h1>

CLOSE (); - выйти из программы.

CLOSEALL (SAVE : save); - сохранить всё и закрыть все открытые проекты и документы.
save -	Y или N -	сохранять при этом проекты и документы

CREATEDOCTEXT (filename); - создать новый документ из текстового файла. При этом он становится текущим документом.
	filename -	строка -	имя файла

CREATEDOCNEW (code, ver, storage); - создать новый документ. При этом он становится текущим документом.
	code - строка - 	код документа
	ver - 	целое число -	версия документа
	storage - строка - 	название хранилища

CREATEDOCSARPS (filename) - создать справочники из файла АРПС 1.10.
filename -	строка -	полное имя файла

CREATEPORTFOLIOFILE (); - создать новый портфель проектов из текстовых файлов формата SP. При этом он становится текущим проектом.

CREATEPORTFOLIONEW (code , ver , storage); - создать новый портфель проектов (при этом он становится текущим проектом).
code -		строка -	код портфеля проектов
ver -		целое число -	версия портфеля проектов
storage -	строка -	название хранилища

CREATEPROJARPS (filename, code, ver) - создать проект из файла АРПС 1.10
filename -	строка -	полное имя файла
code -		строка -	код проекта
ver -		целое число -	версия проекта

CREATEPROJFILE (); - создать новый проект из текстовых файлов формата SP. При этом он становится текущим проектом.

CREATEPROJMSPXML (filename); - создать новый проект из файла MS Project XML.
filename -	строка -	полное имя файла

CREATEPROJNEW (code, ver, storage); - создать новый проект. При этом он становится текущим проектом.
	code - 	строка - 		код проекта
	ver - 		целое число	-	версия проекта
	storage - 	строка - 		название хранилища

CREATEPROJPRMXER (filename); - создать новый проект из файла Primavera XER. 
filename -	строка -	полное имя файла

CREATESTORAGE (name, path, parent) - создать хранилище.
name -	строка -	имя хранилища
path -	строка –	путь
parent - строка -	имя родителя хранилища

DELETEDOC (code, ver, storage); - удалить документ из хранилища.
	code -		строка -	код документа
	ver -		целое число -	версия документа
	storage -	строка -	название хранилища

DELETEPROJ (code, ver, storage); - удалить проект из хранилища.
	code -		строка -	код проекта
	ver -		целое число -	версия проекта
	storage -	строка -	название хранилища

DELETESTORAGE (name); - удалить хранилище.
	name -		строка -	имя хранилища

DIAGRAM (diagram); - делать отчёты в виде диаграммы или табличного документа. 
diagram -	Y или N -	в виде диаграммы (Y) или табличного документа (N)

EXECUTE (val, sync); - запуск внешнего приложения. 
val -	строка –	приложение
sync -	Y или N -	выполнять синхронно или асинхронно

OPENDOC (code, ver, storage); - открыть документ. При этом он становится текущим документом.
	code - 	строка - 		код документа
	ver - 		целое число	-	версия документа (-1 - последняя версия)
	storage - 	строка - 		название хранилища

OPENFILE (filename); - открыть файл (при этом он становится текущим проектом или документом). 
filename -	строка -	полное имя файла

OPENPROJ (code, ver, storage); - открыть проект. При этом он становится текущим проектом.
	code - 	строка - 		код проекта
	ver - 		целое число	-	версия проекта (-1 - последняя версия)
	storage - 	строка - 		название хранилища

SAVEALL (); - сохранить всё.

SETADDCOLUMNDOC (ADD : add, UNIQ : uniq) - установить опции добавления колонок из других документов.
add -	Y или N -	создавать ли новые строки. 
uniq -	Y или N -	при совпадении кодов колонок генерировать уникальные
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETARPS (WORKLOAD : workload, PHASE : phase, COEF : coef, UNIT : unit, DOC : doc, USERFIELDS : userfields) - установить опции создания справочников и проектов из файла АРПС 1.10.
workload -	Y или N -	считать позиции без трудоемкости подчиненными позициями 
(материалами)
phase -	Y или N -	cоздавать фазы по разделам сметы 
coef -		Y или N -	пересчитывать ли значения с коэффициентами
unit -		Y или N -	единица объема операций согласно смете
doc -		Y или N -	создавать ли справочники
userfields -	Y или N -	создавать пользовательские поля с коэффициентами 
накладных расходов и сметной прибыли
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETCOPYPROJ (CMULT : cmult,
             CFROM : cfrom,
             CREPL : crepl,
             CPREF : cpref,
             CPOST : cpost); - установить опции изменения характеристик операций и фаз при копировании 
	cmult - 	дробное число - 	коэффициент умножения объёмов, длительностей и 						расходов копируемых операций
	cfrom -	целое число	- 	с какого символа заменять коды копируемых 							операций и фаз
	crepl -	строка - 		на что заменять коды копируемых операций и фаз
	cpref -	строка - 		какой префикс добавить к кодам копируемых 							операций и фаз
	cpost -	строка - 		какой постфикс добавить к кодам копируемых 							операций и фаз
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETCURRDOC (code, ver, storage); - установить текущий документ.
	code - 	строка - 		код документа
	ver - 		целое число	-	версия документа
	storage - 	строка - 		название хранилища

SETCURRPROJ (code, ver, storage); - установить текущий проект.
	code - 	строка - 		код проекта
	ver - 		целое число	-	версия проекта
	storage - 	строка - 		название хранилища

SETDISTRPROJ (RES : res,
              MAT : mat,
              COST : cost,
              CALEN : calen,
              OPTIONS : options); - установить опции копирования подпроектов.
	res - 		Y или N - 	удалять ли неиспользуемые ресурсы, 								мультиресурсы, роли ресурсов
mat -		Y или N -	удалять ли неиспользуемые материалы и комплекты 
материалов	
	cost -		Y или N -	удалять ли неиспользуемые 										стоимостные составляющие 		
	calen -	Y или N -	удалять ли неиспользуемые 										календари, недели, исключения
	options -	Y или N -	копировать ли настройки расчётов
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETDOCAPPLY (ADD : add, FILTER : filter, TYPE : type, EDIT : edit, TOFILTER : tofilter) - установить опции применения документа.
	add -		Y или N -	создавать ли новую строку, если строка не 							существует
filter -	Y или N -	применять ли только отфильтрованную часть документа 
или весь документ целиком
type -		строка -	тип создаваемых колонок
edit -		Y или N -	редактировать ли существующие строки
tofilter -	Y или N -	применять ли документ только к отфильтрованной части
таблицы или ко всей целиком
Аргумент type может принимать следующие значения: 
TEXT -	тип Текст 
NUM -	тип Число 
TIME -	тип Дата-Время
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETFILEEXPIMP (TEXT : textpath, 
               MONITOR : monitorpath, 
               DELIMITER : delimiter,
               VISIBLEFIELDS : visiblefields,
               USERFIELDS : userfields,
               COST : cost,
               MAT : mat,
               MATSET : matset,
               CODES : codes,
               TYPES : types,
               NAMES : names,
               MONTYPES : montypes,
               MONNAMES : monnames, 
               ALLNUMBERS : allnumbers,
               UNICODE : unicode, 
               NULL : null,
               DELBLANKROW : delblankrow,
               COLOR : color, 
               DATE : date,
               MAIN : main,
               PHASE : phase, 
               WBS : wbs, 
               CSV : csv,
               INCCODEVER : inccodever,
               CODE : code,
               VER : ver); - установить опции импорта-экспорта.
	textpath - 	строка - 	путь к текстовым файлам
	monitorpath -	строка - 	путь к текстовым файлам учёта
	delimiter -	строка -	разделитель
	visiblefields -	Y или N -	экспортировать из таблицы только 									видимые поля 
	userfields -		Y или N -	экспортировать из таблицы 										пользовательские поля 
	cost -			Y или N -	экспортировать из таблицы поля 									расходов стоимостных составляющих 
	mat -			Y или N -	экспортировать из таблицы поля 									расходов материалов
	matset -		Y или N -	экспортировать из таблицы поля 									расходов комплектов материалов
	codes -		Y или N -	добавлять ли при экспорте таблицы в 								строки коды колонок
	types -		Y или N -	добавлять ли при экспорте таблицы в 								первую строку типы колонок
	names -		Y или N -	добавлять ли при экспорте таблицы в 								строки имена колонок
	montypes -		Y или N -	добавлять ли при экспорте учёта в 								строки типы колонок
	monnames -		Y или N -	добавлять ли при экспорте учёта в 								строки имена колонок
	allnumbers -		Y или N -	экспортировать ли все цифры после 								запятой
unicode -		Y или N -	работать ли с текстом в кодировке Юникод
null -			Y или N -	экспортировать ли пустые значения как нули
	delblankrow -		Y или N -	удалять пустые строки
color -		Y или N -	экспортировать ли цвета строк
date -			Y или N -	вставлять ли текущую дату в имя таблицы при 
экспорте в базу данных
	main -			Y или N -	экспортировать только главную структуру проекта
phase -		Y или N -	экспортировать в качестве кода проекта код 
верхней фазы главной структуры
	wbs -			Y или N -	экспортировать в базу данных только фазы и 
стоимости
csv -			Y или N -	тип файлов CSV (Y) или TXT (N) 
inccodever -	Y или N -	включать ли в имена файлов код и версию проекта                                                                                       
code -			строка -	код проекта
ver -			строка -	версия проекта
Разделитель - символ (например, #) или его код в ANSI-кодировке (например, \35). Символ табуляции задаётся строкой TAB. Символ ‘точка с запятой’ задаётся строкой SEMICOLON. Символ ‘запятая’ задаётся строкой COMMA. Символ ‘пробел’ задаётся строкой SPACE. Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

SETFILTERAPPLY (CLEAR : clear, APPLY : apply); - установить опции применения фильтра.
clear -	Y или N -	отменять ли предыдущий фильтр
apply -	строка -	к чему применять фильтр
Аргумент apply может принимать следующие значения: 
GROUP	-	к группам 
OBJECT	-	к объектам
ASSIGN	-	к назначениям

SETMAINOPTIONS (SHOW : show, USER : user, PASSWORD : password);- установить основные настройки.
show -		Y или N -	отображать окна при выполнении сценария
user -		строка -	код пользователя
password -	строка -	пароль пользователя

SETPROJMERGEOPTIONS (OBJECT : object); - установить опции объединения проектов.
object -	строка -	настройка объединения объектов с одинаковыми кодами. Аргумент object может принимать следующие значения: 
MERGE -	объединить
EXIST -	оставить существующий
NEW -	создать новый. 
Аргументы могут перечисляться в произвольном порядке, необязательно перечислять все.

STORAGESETPROP (name, pcode, val) - задать свойства хранилища.
name -		строка -	имя хранилища
pcode -	строка -	код свойства хранилища
val -		строка -	значение свойства 
Допустимые значения кодов свойств хранилища: 
Path - Путь 
DefProj - По умолчанию для проектов (допустимое значение val: Y - Да)
DefRep - По умолчанию для отчётов (допустимое значение val: Y - Да)
DefRef - По умолчанию для справочников (допустимое значение val: Y - Да)


<h1>Команды документа</h1>
	
DOCADDCOLUMNDOC (doccode, ver, storage, code1, name1, newcode1, ..., codeN, nameN, newcodeN); - добавить колонки из документа. 	
	doccode -	строка -	код документа
	ver -		целое число -	версия документа
	storage -	строка -	название хранилища
	codeI -	строка -	код колонки
	nameI -	строка -	новое имя колонки
	newcodeI -	строка -	новый код колонки

DOCADDROWSDOC (doccode, ver, storage); - добавить строки из документа. 	
	doccode -	строка -	код документа 
	ver -		целое число -	версия документа 
	storage -	строка -	название хранилища

DOCADDFIELD (code, name, type); - добавить поле в документ. 
code -	строка -	код поля
name -	строка -	название поля
type -	строка -	тип поля 
Аргумент type может принимать следующие значения: 
TEXT -	тип Текст
NUM -	тип Число
TIME -	тип Дата-Время

DOCAPPLYCOLUMNS (code, ver, storage); - применить документ к колонкам документа.
	code -		строка -	код документа 
	ver -		целое число -	версия документа 
	storage -	строка -	название хранилища

DOCAPPLYCURRPROJ (); - применить текущий документ к текущему проекту.

	MAT - 			таблица материалов

	MATSET - 		таблица комплектов материалов
	MONTECARLO - 		таблица данных Монте-Карло
	OPER - 		таблица операций
	OVERTIME - 		таблица сверхурочных
	PRINT - 		таблица шаблонов печати
	PROJ - 		таблица реестра портфеля проектов
	PROTOCOL - 		таблица журнала
	REPORT - 		таблица шаблонов отчётов

	USER - 		таблица пользователей

	WEEK - 		таблица недель

DOCAPPLYDOC (code, ver, storage); - применить документ к документу.
	code -		строка -	код документа
	ver -		целое число -	версия документа
	storage -	строка -	название хранилища

DOCAPPLYFILTER (code); - применить фильтр к документу.
	code -	строка -	код фильтра

DOCAPPLYFILTERVALUES (code, condition, value); - применить фильтр по значениям к документу.
code -		строка -	код поля
condition -	строка –	условие
value -	строка -	значение
Аргумент condition может принимать одно из следующих значений: 
EQU –		равно
NOTEQU -	не равно
LESS –		меньше
LESSEQU -	меньше или равно
GREAT –	больше
GREATEQU -	больше или равно
CONTAIN –	содержит
NOTCONTAIN -	не содержит

DOCAPPLYFORMULA (code); - выполнить формулу в документе.
	code -	строка -	код формулы	

DOCAPPLYMULTIROJ (code, ver, storage); - применить документ к портфелю проектов.
	code -		строка -	код портфеля проектов
	ver -		целое число -	версия портфеля проектов
	storage -	строка -	название хранилища

DOCAPPLYPROJ (code, ver, storage); - применить документ к проекту.
	code -		строка -	код проекта
	ver -		целое число -	версия проекта
	storage -	строка -	название хранилища

DOCAPPLYPROJTAB (code, ver, storage, tab); - применить документ к таблице проекта.
	code -		строка -	код проекта
	ver -		целое число -	версия проекта
	storage -	строка -	название хранилища
	tab -		строка -	таблица проекта
Аргумент tab может принимать следующие значения:
	ACTUAL - 		таблица исполнения
	AREA - 		таблица областей для анализа рисков
	ASSIGN - 		таблица назначений ресурсов
	CALEN - 		таблица календарей
	CONFIG - 		таблица конфигураций
	CORPREF - 		таблица корпоративных справочников
	COST - 		таблица стоимостных составляющих
	COSTCENT - 		таблица центров стоимостей
	COSTPERIOD - 		таблица периодов стоимостей
